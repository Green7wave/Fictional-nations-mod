 	title = grn.1.t

 		tag = GRN

 	}

 			limit = {
 				is_owned_by = GRN
 				industrial_complex > 0
 			}
 	
 			damage_building = {
 				type = industrial_complex 
 				damage = 5
 			}
 		}
 		# damage mils and infe
 		every_state = {
 			limit = {
 				is_owned_by = GRN
 				arms_factory > 2
 			}
 	
 			damage_building = {
 				type = arms_factory 
 				damage = 2
 			}
 
 		}
 		# remove some ipliereng factories
 		if {
 			limit = {
 				48 = { 
 					is_owned_by = GRN 
 					industrial_complex > 1
 				}
 			}
 			48 = {
 				remove_building = {
   				  type = industrial_complex
   				  level = 2
 				}
 			}
 		}
 		# removes some of the arms factories
 		if {
 			limit = {
 				212 = { 
 					is_owned_by = GRN 
 					arms_factory > 2
 				}
 			}
 			212 = {
 				remove_building = {
   				  type = arms_factory
   				  level = 3
 				}
 			}
 		}
 		# get the new idea bro *sad Greenlin noises*

 	}
