# The bombs blow up
 country_event = {
 	id = grn.1
 	title = grn.1.t
 	desc = grn.1.d
 	picture = GFX_report_event_generic_ruins
 
 	fire_only_once = yes
 	
 	trigger = {
 		tag = GRN
 		has_war = no
 	}
 	
 	mean_time_to_happen = {
 		days = 10
 	}
 	
 	option = { 
 		name = grn.1.a
 		# damage civs
 		every_state = {
 			limit = {
 				is_owned_by = GRN
 				industrial_complex > 0
 			}
 	
 			damage_building = {
 				type = industrial_complex 
 				damage = 5
 			}
 		}
 		# damage mils and infe
 		every_state = {
 			limit = {
 				is_owned_by = GRN
 				arms_factory > 2
 			}
 	
 			damage_building = {
 				type = arms_factory 
 				damage = 2
 			}
 
 		}
 		# remove some ipliereng factories
 		if {
 			limit = {
 				48 = { 
 					is_owned_by = GRN 
 					industrial_complex > 1
 				}
 			}
 			48 = {
 				remove_building = {
   				  type = industrial_complex
   				  level = 2
 				}
 			}
 		}
 		# removes some of the arms factories
 		if {
 			limit = {
 				212 = { 
 					is_owned_by = GRN 
 					arms_factory > 2
 				}
 			}
 			212 = {
 				remove_building = {
   				  type = arms_factory
   				  level = 3
 				}
 			}
 		}
 		# get the new idea bro *sad Greenlin noises*
 		swap_ideas = {
   			remove_idea = industrial_issue
  			add_idea = industrial_issue2
 		}
         # let the code know we had the event
 		set_country_flag = ling_attacked
 		
 	}
 	
}
